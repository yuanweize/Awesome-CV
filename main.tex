%-------------------------------------------------------------------------------
% CONFIGURATIONS
%-------------------------------------------------------------------------------
\documentclass[11pt, a4paper]{awesome-cv}

% Configure page margins with geometry
\geometry{left=1.4cm, top=1.5cm, right=1.4cm, bottom=1.8cm, footskip=.5cm}

% Color for highlights
% Options: awesome-emerald, awesome-skyblue, awesome-red, awesome-pink, awesome-orange
%          awesome-nephritis, awesome-concrete, awesome-darknight
\colorlet{awesome}{awesome-red}
% Or define your own: \definecolor{awesome}{HTML}{3E6D9C}

% Set false if you don't want to highlight section with awesome color
\setbool{acvSectionColorHighlight}{true}

% Social information separator
\renewcommand{\acvHeaderSocialSep}{\quad\textbar\quad}

% Enlarge header elements for better first-page presence
\renewcommand*{\headerpositionstyle}[1]{{\fontsize{10pt}{1em}\bodyfont\scshape\color{awesome} #1}}
\renewcommand*{\headeraddressstyle}[1]{{\fontsize{10pt}{1em}\headerfont\itshape\color{lighttext} #1}}
\renewcommand*{\headersocialstyle}[1]{{\fontsize{10pt}{1em}\headerfont\color{text} #1}}
\renewcommand*{\headerquotestyle}[1]{{\fontsize{10pt}{1em}\bodyfont\itshape\color{darktext} #1}}


%-------------------------------------------------------------------------------
% PERSONAL INFORMATION
% Load from config.tex (gitignored - contains real PII)
%-------------------------------------------------------------------------------
\input{config.tex}


%-------------------------------------------------------------------------------
% CONTENT PATH CONFIGURATION
% Change this to switch between template and real content
%-------------------------------------------------------------------------------
% For CI/demo: use sections_template
% For personal use: use sections (gitignored)
\newcommand{\contentpath}{sections}

% Fallback: check if sections/ exists, otherwise use sections_template/
\IfFileExists{sections/summary.tex}{
  \renewcommand{\contentpath}{sections}
}{
  \renewcommand{\contentpath}{sections_template}
}


%-------------------------------------------------------------------------------
\begin{document}

% Print the header with above personal information
% Alignment options: C (center), L (left), R (right)
\makecvheader[C]

% Print the footer
\makeatletter
\makecvfooter
  {\today}
  {\@firstname~\@lastname~~~·~~~Résumé}
  {\thepage}
\makeatother


%-------------------------------------------------------------------------------
% CV/RESUME CONTENT
% Each section is imported separately
%-------------------------------------------------------------------------------
\input{\contentpath/summary.tex}
\input{\contentpath/education.tex}
\input{\contentpath/skills.tex}
\newpage
\input{\contentpath/experience.tex}
\input{\contentpath/certificates.tex}
\input{\contentpath/honors.tex}


%-------------------------------------------------------------------------------
\end{document}
